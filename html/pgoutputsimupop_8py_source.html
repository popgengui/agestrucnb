249  def bz2CompressAllFiles(self, ls_files_to_skip=[] ):
250  '''
251  used code and advice in,
252  http://stackoverflow.com/questions/9518705/big-file-compression-with-python
253 
254  Note: checked the shutil documentation at https://docs.python.org/2/library/shutil.html
255  which warns of loss of meta file info (owner/group ACLs) when using shutil.copy() or shutil.copy2(). Unclear
256  whether this applies to the copyfileobj, though my few tests showd all of these were retained.
257  '''
258  for s_myfile in [ self.__outname, self.__errname, self.__megadbname, self.__confname, self.__genepopname ]:
259 
260  if s_myfile in ls_files_to_skip:
261  pass
262  elif not self.__file_exists( s_myfile ):
263  self.__raise_file_not_found_error( s_myfile, "compress file with bz2" )
264  else:
265  with open( s_myfile, 'rb' ) as o_input:
266  with bz2.BZ2File( s_myfile + '.bz2', 'wb', compresslevel=9 ) as o_output:
267  #try/except overkill, but want logic to show
268  #we don't remove the input file
269  #if the copy op fails:
270  try:
271  shutil.copyfileobj( o_input, o_output )
272  except Exception as e:
273  raise e
274  #end try except
275 
276  #required by windows only, else error can't remove:
277  o_input.close()
278  os.remove( s_myfile )
279  #end with bzfile for writing
280  #end with current file for reading
281  #end if file absent, else exists
282  #end for each file
283  #end bz2CompressAllFiles

335 
336  '''
337  reads the *.gen file from the simuPop output
338  and produces a genepop file based on the gen file
339 
340  param idx_allele_start: int, one-based index giving the
341  item (loci) number of the first allele entry in the line of gen-file input
342  Example, if the *gen file has 10 microsates and 40 SNPs, then its first
343  10 allele entries (that follow the indiv number and the gneration number)
344  will be the microsats, and the final 40 allele entries will be the SNPs
345  If we wanted only the Microsats to be included in the gen file, then
346  this param would be "1", and the idx_allele_stop value would be 10.
347  If we want both msats and snps, we'd give 1 and 50 for these params.
348  If we want only SNPs, wed enter 11 for start allele and 50 for stop.
349 
350  param idx_allele_stop: int, one-based index giving the
351  item (loci) number of the last allele entry in the line of gen-file input
352 
353  param optional b_do_compress, if true, will bzip2 the output genepop file
354 
355  param optional b_pop_per_gen, if true, will insert a "Pop" line between the
356  generations, as given by the 2nd field in the gen file
357 
358  2017_02_12. In implementing a bias adjustment in the NeEstimator's ldne-based
359  Nb estimates, calculated in pgdriveneestimator.py, We add to this def an arg
360  "f_nbne_ratio", defalut None. If arg is non-None, we append to the header line
361  text, "nbne=str(f_nbne_ratio)". When the GUI interface for the Nb estimation reads
362  in genepop files, it will check for they key=value string and then, if present,
363  pass the value on to the calls that eventually invoke the pgdriveneestimator.
364  '''
365 
366  o_genfile=None
367 
368  o_genepopfile=None
369 
370  '''
371  2017_04_21. We change the way we name the temp file, so that it will be written
372  in the output base path, rather than the currdir. Note that we define the var
373  here, but create the name just before using, as the check for duplicate file names,
374  which is perfomred in the call to our new def __get_temp_file_name,
375  should be done just before we then open the file.
376  '''
377  s_temp_file_name=None
378 
379  s_genepop_filename=self.__genepopname
380 
381  i_num_loci=(idx_allele_stop-idx_allele_start) + 1
382 
383  IDX_GEN_INDIV=0
384  IDX_GEN_GENERATION=1
385  IDX_FIRST_ALLELE=2
386 
387  i_genepop_already_exists=self.__file_exists( s_genepop_filename )
388 
389  if i_genepop_already_exists != FILE_DOES_NOT_EXIST:
390 
391  self.__raise_file_exists_error( s_genepop_filename,
392  "create genepop file from gen file" )
393  #end if genepop file already exists
394 
395  i_genfile_exists=self.__file_exists( self.__genfile )
396 
397  #if we have the uncompressed version available
398  #we use it:
399  if i_genfile_exists == FILE_EXISTS_UNCOMPRESSED \
400  or i_genfile_exists == FILE_EXISTS_AS_BOTH_UNCOMPRESSED_AND_BZ2:
401 
402  o_genfile=open( self.__genfile )
403 
404  elif i_genfile_exists == FILE_EXISTS_AS_BZ2:
405  o_genfile=bz2.BZ2File( self.__genfile + ".bz2", mode='r' )
406  else:
407  self.__raise_file_not_found_error( self.__genfile, "convert gen file to genepop" )
408  #end if uncompressed only or uncomp. and compressed, else compressed only, else no file
409 
410  s_temp_file_name=self.__get_temp_file_name()
411 
412  if b_do_compress == True:
413  o_genepopfile=bz2.BZ2File( s_temp_file_name + '.bz2', 'w', compresslevel=9 )
414  else:
415  o_genepopfile=open( s_temp_file_name, 'w' )

420  i_num_loci,
421  f_nbne_ratio,
422  b_do_compress )
423 
424  #write indiv id and alleles, line by line:
425  i_currgen=None
426 
427  for s_line in o_genfile:
428 
429  '''
430  2017_05_02. For python 3, because in a compressed file python 3 reads bytes
431  instead of strings (despite opening bzw2 in in mode='r'),
432  we need to decode if we read in bytes.
433  '''

435  s_line=s_line.decode( SYSDEFAULTENCODING )

488

490 
491  #end for each line in gen file
492 
493  o_genfile.close()
