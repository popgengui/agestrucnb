389  def parseNoDatFile( self ):
390  '''
391  This def is provisional, since I have only seen
392  a few of the *NoDat.txt files generated by
393  NeEstimator (v2). I base the parsing on examples like this:
394 
395  Population 1 [OmyLGRA12S_0213]
396  -----------------------------------------------------------
397  Individual Locus Genotype Number of Loci
398  with missing data
399  1 9 0000 4
400  ----------------------------------------------------------
401 
402 
403 
404  However thin the ground, this def assumes
405 
406  1. multipop results will simply have more table entries
407  like the one above.
408 
409  2. If a line in the NeEstimator's "*NoDat.txt" file
410  starts with the value below in COLHEADSTART, and
411  contains the other column keywords given below
412  in COLSINCLUDE, that is must be the header and
413  also that the data follows after one mostly line blank
414  but for a tag on the last column header
415 
416  3. Population name entry for each table is uniqely
417  located by the line starting with value below in POPLINESTART
418  '''
419 
420 
421  POPLINESTART="Population"
422  COLHEADSTART="Individual"
423  COLSINCLUDE=[ "Locus", "Genotype", "Number of Loci" ]
424  ENDTABLELINESTART="----------------------"
425 
426  s_name=self.__get_name_nodat_file()
427 
428  if s_name is None:
429  return
430  #end return if no nodat file
431 
432  self.__parsed_nodat_output=[]
433 
434  o_file=open( s_name )
435 
436  b_found_header=False
437  i_table_lines_count=0
438  s_currpop=""
439  for s_line in o_file:
440 
441  i_num_match=0
442  if s_line.startswith( POPLINESTART ):
443  #we replace spaces in the pop name:
444  s_currpop=s_line.strip().replace( " ", "_" )
445  elif s_line.startswith(COLHEADSTART):
446  lb_trues=[ s_col in s_line for s_col in COLSINCLUDE ]
447  if sum( lb_trues ) == len( COLSINCLUDE ):
448  b_found_header=True
449  i_table_lines_count=0
450  #end if all header names in line
451  else:
452  #first line after header is more header:
453  if b_found_header and i_table_lines_count > 1:
454  #if end of table for currpop,
455  #reset flag, line counts
456  if s_line.startswith( ENDTABLELINESTART ):
457  b_found_header=False
458  i_table_lines_count=0
459  else:
460  i_table_lines_count+=1
461  #NoDat.txt files data cols
462  #are seperated by multiple space chars,
463  #so we use the default split to
464  #list only non-space as items:
465  ls_vals=s_line.strip().split()
466  #first col is the pop name
467  self.__parsed_nodat_output.append( [ s_currpop ] \
468  + [ s_val for s_val in ls_vals ] )
469  #end if end-table line, else data
470  #end if header's been found and we're past the non-date line
471  #end if pop line else header line, else some other line
472  #end for each line in nodat file
473 
474  o_file.close()
475  return
476  #end parseNoDatFile
