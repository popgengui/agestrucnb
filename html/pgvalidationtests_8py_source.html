2 '''
3 Description
4 We needed a suite of tests to validate the simulation and nb output.
5 The first of these requested by Gordon is a comparison of expected
6 hegerozygosity, calculated from the Ne value, for the tth generation
7 as ( 1-2/Ne )**t, vs the "observed"
8 loss of heterozygosity, calculated for each generation as the mean ( 1-He ), with
9 He_l = sum_i_l=1 to n_l (i_l**2), with i_l=allele freq of the ith of n alleles
10 for the lth loci.
11 
12 
13 2017_02_22
14 We've clarified that we will add a cycles/generation ratio to our input values
15 in order to correctly implement the theoretical vs expected values. Note that
16 plotting for most of our model species by sim breeding cycles, implying a 1/1
17 breeding cycle per generation ratio for measuring heterozygosity, shows
18 inaccurate declines, expected vs theoretical.
19 '''

142 def get_theoretical_heterozygosity_per_generation( df_mean_het_by_cycle,
143  df_ne_by_cycle,
144  f_cycles_per_generation,
145  f_theoretical_ne=None ):
146  f_reltol=1e-80
147 
148  '''
149  From Brian H, the iterative caluclation:
150  H_t = H_[t-1](1 - 1/(2Ne_0)), where Ne_0 is the Ne value for the
151  initial population.
152  '''
153 
154  df_theoretical_het_per_generation={}
155 
156  li_cycle_numbers_for_ne=list(df_ne_by_cycle.keys())
157  li_cycle_numbers_for_het=list(df_mean_het_by_cycle.keys())
158 
159  li_cycle_numbers_for_ne.sort()
160  li_cycle_numbers_for_het.sort()
161 
162  i_cycles_per_generation=int( round( f_cycles_per_generation ) )
163 
164  if li_cycle_numbers_for_ne != li_cycle_numbers_for_het:
165 
166  s_msg="In pgvalidationtests.py, " \
167  + "def get_theoretical_heterozygosity_per_generation, " \
168  + "Ne values dict has keys (cycle numbers) " \
169  + "that do not equal those in the Het values dict. " \
170  + "Ne dict keys: " + str( list( df_ne_by_cycle.keys() ) ) \
171  + ", Het dict keys: " + str( list( df_mean_het_by_cycle ) ) \
172  + "."
173  raise Exception( s_msg )
174  #end if non-equal key sets
175 
176  i_total_cycle_numbers=len( list(df_ne_by_cycle.keys()) )
177  i_lowest_cycle_number=min( df_ne_by_cycle.keys() )
178 
179  f_ne0=f_theoretical_ne if f_theoretical_ne is not None \
180  else np.mean( list( df_ne_by_cycle.values() ) )
181 
182  f_ht0=df_mean_het_by_cycle[ i_lowest_cycle_number ]
183 
184  '''
185  The initial generation has the het
186  value caluclated by mean sum of squared
187  allele frequencies over all loci, so
188  that for gen 1, the theoretical and the
189  expected match:
190  '''
191  i_generation_number= 1
192 
193  df_theoretical_het_per_generation[ 1 ] = f_ht0
194 
195  f_calc_coefficient = 1 if f_ne0 <= f_reltol else 1 - old_div(1, ( 2*f_ne0 ))
196 
197  '''
198  We use cycles per generation to
199  itereate the correct number of times in the calculation
200  '''
201 
202 
203  for idx in range ( i_cycles_per_generation, i_total_cycle_numbers, i_cycles_per_generation ):
204 
205 
206  i_generation_number+=1
207  i_this_cycle_number=li_cycle_numbers_for_ne[ idx ]
208 
209  f_het_value_last_generation=df_theoretical_het_per_generation[ i_generation_number - 1 ]
210 
211  f_theoretical_het_for_this_generation = f_het_value_last_generation * f_calc_coefficient
212 
213  df_theoretical_het_per_generation[ i_generation_number ] = \
214  f_theoretical_het_for_this_generation
215  #end for each index into list cycle numbers
216 
217  return df_theoretical_het_per_generation
218 #end get_theoretical_heterozygosity_per_generation

221 def get_theoretical_het_loss_per_generation( df_ne_by_cycle_number, f_cycles_per_generation, f_theoretical_ne = None ):
222  '''
223 
224  This calculation for theoretical loss of heterozygosity is from
225  From http://labs.russell.wisc.edu/peery/files/2011/12/Primer-in-Population-Genetics.pdf
226 
227  This def also returns the Ne value for the initial cycle.
228  '''
229 
230  df_theoretical_het_loss_by_generation={}
231 
232  i_cycles_per_generation=int( round( f_cycles_per_generation ) )
233 
234  li_sorted_cycle_numbers=list(df_ne_by_cycle_number.keys())
235  li_sorted_cycle_numbers.sort()
236 
237  i_min_gen_number=min( li_sorted_cycle_numbers )
238 
239  f_ne_for_initial_pop=f_theoretical_ne if f_theoretical_ne is not None \
240  else np.mean( list( df_ne_by_cycle_number.values() ) )
241 
242  f_one_over_2Ne=old_div(1.0,( 2.0*f_ne_for_initial_pop ))
243 
244  f_one_minus_one_over_2Ne=1 - f_one_over_2Ne
245 
246  '''
247  We want zero loss of het in the
248  first cycle, so the counter,
249  which gives the exponent in the
250  loss calculation, will be at zero
251  for the first (initial) cycle.
252  '''
253  i_generation_count = -1
254 
255  i_generation_number=0
256 
257  for i_gen in range( 0, len( li_sorted_cycle_numbers ), i_cycles_per_generation ):
258 
259  i_generation_count += 1
260  i_generation_number+=1
261 
262  '''
263  We raise ( 1 - 1/2Ne ) to t = gen_counter, subtract
264  that value from 1:
265  '''
266  f_loss_this_gen=1 - ( ( f_one_minus_one_over_2Ne )**i_generation_count )
267 
268  df_theoretical_het_loss_by_generation[ i_generation_number ] = f_loss_this_gen
269  #end for each cycle
270 
271  return df_theoretical_het_loss_by_generation
272 
273 #end get_theoretical_het_loss_per_generation

489 def get_expected_vs_observed_heterozygosity( s_ne_tsv_file,
490  i_initial_cycle_number,
491  i_final_cycle_number,
492  f_cycles_per_generation,
493  i_min_loci_number=None,
494  i_max_loci_number=None,
495  f_theoretical_ne=None,
496  b_plot=True,
497  s_plot_file=None,
498  s_plot_type="png" ):
499 
500  '''
501  This def is designed to use either a user-supplied theoretical Ne, or to
502  use the mean estimates in the tsv file for the theoretical Ne, and, for
503  the initial heterozygosity the mean HW expected via allele frequencies
504  for the first (initial) cycle. The theoretical het for the ith generation
505  is calculated as
506  Hz_i=Hz_{i-1} X ( 1 - 1/(2Ne) )
507 
508  For the expected het for each cycle:
509  sum_1_to_L( 1 - sum_1_to_A( allele-frequency_ij ** 2 ) )/L, in a cycle
510  (pop section) with L loci, and each loci with some total alleles A, i=ith loci,
511  j=jth allele.
512 
513  Assumption: that the (genepop) file(s) named in column one of the s_ne_tsv_file
514  argument exist and have either absolute paths (the current
515  default method used to write them to the tsv file), or relative
516  paths that are accessible from the working directory of the
517  current python environment.
518 
519  param s_ne_tsv_file, names the table file output from using module
520  pgdriveneestimator.py to generate ne estimations, either
521  via the negui interface or the command line use of pgdriveneestimator.py
522  itself, on genepop files generated by the negui implementation of
523  the AgeStructureNe simulation driver.
524  param i_initial_cycle_number, integer giving the ith pop section,
525  where i is the poplulation (cycle) that gives the initital Ne value on which
526  to base the expected loss of het, that is the Ne value in ( 1-2/Ne )^t for
527  each subsequent cycle t.
528  param i_final_cycle_number, if not None, then an integer giving the highest pop section
529  (cycle) value to analyze. If None, then analyze all pop sections with numbers
530  >= i_initial_cycle_number
531  param f_cycles_per_generation. gives the number of breeding cycles per generation, so
532  that theoretical interations are done per generation (not per cycle, unless
533  cycles per generation is 1.0
534  param i_min_loci_number, if not none than the He calculation will not use any of the loci
535  in the range 1 to i, where i_min_loci_number is i + 1.
536  param i_ma_loci_number, if not none than the He calculation will not use any of the N loci
537  in the range j to N, where i_man_loci_number is j - 1.
538 
539  param b_plot, if true and maplotlib was importable, then a plot will be done.
540  param s_plot_file, if not none and b_plot is true, then plot to file with this name.
541  If s_plot_file is None, then call pyplot.show() after the plot command.
542  param s_plot_type, if b_plot is true and s_plot_file is not None, the file type
543  to be written is given by this extension, defaulting to png.
544 
545  '''
546 
547 
548  ddf_mean_het_by_gp_file_by_cycle, ddf_ne_by_gp_file_by_cycle = \
549  get_mean_hets_and_ne_for_cycles( s_ne_tsv_file,
550  i_initial_cycle_number,
551  i_final_cycle_number,
552  i_min_loci_number,
553  i_max_loci_number )
554 
555  for s_gp_file_name in ddf_mean_het_by_gp_file_by_cycle:
556 
557  df_het_per_cycle=ddf_mean_het_by_gp_file_by_cycle[ s_gp_file_name ]
