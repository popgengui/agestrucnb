194  def __scrollCanvas( self, event ):
195  '''
196  Platform-dependant handling according to,
197  http://stackoverflow.com/questions/17355902
198  /python-tkinter-binding-mousewheel-to-scrollbar
199  also, used the linux handling code from:
200  https://mail.python.org/pipermail/tkinter-discuss
201  /2006-June/000808.html.
202  '''
203  s_platform=pgut.get_platform()
204  if s_platform == pgut.SYS_LINUX:
205  if event.num == 4:
206  self.__canvas.yview( 'scroll', -1, 'units' )
207  elif event.num== 5:
208  self.__canvas.yview( 'scroll', 1, 'units' )
209  #end if event num is 4 else 5
210  elif s_platform == pgut.SYS_MAC:
211  self.__canvas.yview_scroll( 1 * event.delta, 'units' )
212  elif s_platform == pgut.SYS_WINDOWS:
213  self.__canvas.yview_scroll( -1 * ( old_div(event.delta,120) ), 'units' )
214  #end if linux, else mac, else windows
215  return
