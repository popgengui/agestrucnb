162  s_chrom=ls_fields[ myc.IDX_CHROM_NAME ]

167  '''
168  In our example input file, we see loci with distinct
169  name but identical chromosome and position. Although
170  for simuPOP's purposes, such loci are indistinguishable,
171  but we preserve this ambiguity, as we can resove these
172  by "jiggling" the position just enougth (1e-15)
173  for simupop to allow them to be input as separate loci
174  (see def below in class PGSimupopLociInfo):
175  '''
176  if f_position not in self.__loci_name_by_position_by_chrom[ s_chrom ]:
177  self.__loci_name_by_position_by_chrom[ s_chrom ][ f_position ]=[ s_locus_name ]
