302  def __looks_like_tsv_file( self, s_tsv_file_name ):
303  '''
304  This def is a heuristic, very approximate,
305  for deciding whether is it likely that the file is
306  a valid *tsv file as generated by our NeEstimation
307  functions. We use the header and test forthe column
308  names that this class uses to group and filter
309  the data.
310  '''
311 
312  b_verdict=True
313 
314  o_myc=PGNeEstimationBoxplotInterface
315  o_tsvc=NeEstimationTableFileManager
316 
317  ls_column_names_to_test=o_myc.GROUP_BY_COLUMNS + o_myc.Y_AXIS_VALUE_COLUMNS
318 
319  o_tsv=open( s_tsv_file_name, 'r' )
320  s_header=o_tsv.readline()
321  o_tsv.close()
322 
323  ls_header_fields = s_header.strip().split( o_tsvc.DELIM_TABLE )
324 
325  for s_name in ls_column_names_to_test:
326 
327  if s_name not in ls_header_fields:
328  b_verdict=False
329  break
330  #end if column name missing
331  #end for each column name to test
332 
333  return b_verdict
334  #end __looks_like_tsv_file

336  def __looks_like_a_tsv_file_with_data_lines( self, s_file_name ):
337  '''
338  Another simple heuristic added to the header check
339  in __looks_like_tsv_file, to avoid loading
340  empty *tsv files i.e. header is correct
341  but no data lines follow.
342  '''
343  b_return_value=False
344 
345  if self.__looks_like_tsv_file( s_file_name ) == True:
346  o_tsvc=NeEstimationTableFileManager
347 
348  o_file=open( s_file_name, 'r' )
349  #read the header:
350  s_header=o_file.readline()
351  ls_vals=s_header.split( o_tsvc.DELIM_TABLE )
352  i_num_fields=len( ls_vals )
353  ls_second_line=o_file.readline()
354 
355  if ls_second_line is not None:
356  ls_second_line_values=ls_second_line.split( o_tsvc.DELIM_TABLE )
357  i_num_values=len( ls_second_line_values )
358  if i_num_values == i_num_fields:
359  b_return_value=True
360  #end if value count equals field count
361  #end if we have a second line
362  #end if the file looks like a tsv file
363 
364  return b_return_value
365  #end __looks_like_a_tsv_file_with_data_lines
