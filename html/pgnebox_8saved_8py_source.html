219  def __looks_like_tsv_file( self, s_tsv_file_name ):
220  '''
221  This def is a heuristic, very approximate,
222  for deciding whether is it likely that the file is
223  a valid *tsv file as generated by our NeEstimation
224  functions. We use the header and test forthe column
225  names that this class uses to group and filter
226  the data.
227  '''
228 
229  b_verdict=True
230 
231  o_myc=PGNeEstimationBoxplotInterface
232  o_tsvc=NeEstimationTableFileManager
233 
234  ls_column_names_to_test=o_myc.GROUP_BY_COLUMNS + o_myc.Y_AXIS_VALUE_COLUMNS
235 
236  o_tsv=open( s_tsv_file_name )
237  s_header=o_tsv.readline()
238  o_tsv.close()
239 
240  ls_header_fields = s_header.split( o_tsvc.DELIM_TABLE )
241 
242  for s_name in ls_column_names_to_test:
243 
244  if s_name not in ls_header_fields:
245  b_verdict=False
246  break
247  #end if column name missing
248  #end for each column name to test
249 
250  return b_verdict
