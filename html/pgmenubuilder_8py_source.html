20 class PGMenuBuilder(object):
21  '''
22  Instantiates a tkinter menu_bar, with parent gui object (usually the main window)
23  as given by __init__ param, and populates it by reading a list of menu items and
24  parameters from a configuration file, also an __init__ param. The menus get their
25  command callback defs, as named in the configuration file, in the class object
26  passed as init param o_pg_gui_app.
27 
28  The configuration file key=value entries inside each [section] are of two classes:
29 
30  1. keys that are not param keywords for the call to menu.add_command. These include
31  (a) "parent=value", which tells the
32  builder which of the file labels (sections in the config file) is the parent menu
33  to it's submenu. As of 20160116 this is not implemented, and only parent=None (unquoted)
34  is allowed.
35  (b) "menu_underline=value", where value is an (zero-based) index into the menu label name
36  (as given by the section name).
37 
38  2. key=value pairs whose keys are param keywords (like "accelerator"). Keys must match exactly
39  prarm names in the menu.add_command definition's param list. Their values should be
40  entered in square brackets like a python list, with one entry for each of the menu items.
41  If the nth item is to be "seperator" item, the nth item in the label list should be
42  "sep", and other items like "command" or accelerator, can be any value (ex: None),
43  as they will be ignored.
44 
45  For example, a menubar with 2 items "File" and "Edit", each with 2 menu items, with
46  Edit having a seperator between its items would look like this:
47 
48  #file menu
49  [File]
50  parent=None
51  #which characters to underline in the File label:
52  menu_underline=0
53  label=[ "New" , "Open" ]
54  #which characters to underline in the menu items:
55  underline=[ 0, 0 ]
56  accelerator=[ "Ctrl+N","Ctrl+O" ]
57  command=[ "new_file", "open_file" ]
58 
59  #edit menu
60  [Edit]
61  parent=None
62  menu_underline=0
63  label=["Undo","sep","Options"]
64  underline=[ 1,None, 1 ]
65  accelerator=["Ctrl+Z", None, "Ctrl+T"]
66  command=[ "undo", None, "showopts" ]
67  '''

130  def get_add_command_params_and_args( self, o_config, s_menu_label ):
131  '''
132  returns a list of dictionaries, one for each menu item, of the param=value
133  settings for calls to menu.add_command. We skip the config key=value for keys
134  that are not part of the params needed for add_command.
135  '''
136  l_keyword_menu_params = o_config.options( s_menu_label )
137 
138  #how many items on this menu should equal number of labels:
139  i_num_items = self.__get_num_labels_menu( o_config, s_menu_label )
140 
141  ld_args = [ {} for idx in range( i_num_items ) ]
142 
143  #for each submenu item, we get a dict of keyword:value to pass to "add_command"
144  for s_keyword in l_keyword_menu_params:
145  s_vals=o_config.get( s_menu_label, s_keyword )
146 
147  #for each keyword that is a param name in the menu.add_command call
148  if s_keyword not in PGMenuBuilder.CONFIG_OPTIONS_NON_COMMAND_ARGS:
149 
150  l_vals=eval( s_vals )
151 
152  if len( l_vals ) != i_num_items:
153  raise Exception( "in pgmenubuilder, config entry should have " \
154  + "one item for each label, but the option with keyname, " \
155  + s_keyword + " does not." )
156  #end if wrong number values
157 
158  #if this is the list of command values (callback defs)
159  #for each submenu item,
160  #then we must be able to find it among the attributes
161  #of our member operation object
162  #note: if "sep" is the label, we set other param values
163  #to None, but they are ignored (see for loop below)
164  if s_keyword =="command":
165  l_vals = [ getattr( self.__pg_gui_app, thisval ) \

167  else None for thisval in l_vals ]
168  #end if command list
169

171  ld_args[ idx ][ s_keyword ] = l_vals[ idx ]
172  #end for each item
173  #end if not a param keyword for add_command
174  #end for each keyword

218  def __add_menu_item( self, o_config, s_menu_label ):
219 
220  #this will be the menu labeled s_menu_label (ex: "File" menu)
221  #and the command items (like "Open" or "Save""Save") are then added
222  #to this item using the o_config objects options:
223  o_this_menu = Menu( self.__menubar, tearoff = self.__tearoff )
224 
225  ld_args = self.get_add_command_params_and_args( o_config, s_menu_label )
226 
227  for d_args in ld_args:
228  if d_args[ "label" ] == "sep":
229  o_this_menu.add_separator()
230  else:
231  o_this_menu.add_command( **d_args )
232  #end if seperator, else sub menu
233  #end for each set of args
234 
235  dv_menu_values = self.__get_menu_values( o_config, s_menu_label )
236 
237  s_accel = dv_menu_values[ "accelerator" ] if dv_menu_values[ "accelerator" ] is not None else None
238 
239  self.__menubar.add_cascade( label = s_menu_label,
240  menu = o_this_menu,
