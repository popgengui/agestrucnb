195  def __scrollCanvas( self, event ):
196  '''
197  Platform-dependant handling according to,
198  http://stackoverflow.com/questions/17355902
199  /python-tkinter-binding-mousewheel-to-scrollbar
200  also, used the linux handling code from:
201  https://mail.python.org/pipermail/tkinter-discuss
202  /2006-June/000808.html.
203  '''
204  s_platform=pgut.get_platform()
205  if s_platform == pgut.SYS_LINUX:
206  if event.num == 4:
207  self.__canvas.yview( 'scroll', -1, 'units' )
208  elif event.num== 5:
209  self.__canvas.yview( 'scroll', 1, 'units' )
210  #end if event num is 4 else 5
211  elif s_platform == pgut.SYS_MAC:
212  self.__canvas.yview_scroll( old_div(event.delta,120) , 'units' )
213  elif s_platform == pgut.SYS_WINDOWS:
214  self.__canvas.yview_scroll( -1 * ( old_div(event.delta,120) ), 'units' )
215  #end if linux, else mac, else windows
216  return
