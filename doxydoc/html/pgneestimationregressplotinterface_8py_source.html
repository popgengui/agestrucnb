321  def __looks_like_tsv_file( self, s_tsv_file_name ):
322  '''
323  This def is a heuristic, very approximate,
324  for deciding whether is it likely that the file is
325  a valid *tsv file as generated by our NeEstimation
326  functions. We use the header and test forthe column
327  names that this class uses to group and filter
328  the data.
329  '''
330 
331  b_verdict=True
332 
333  o_myc=PGNeEstimationRegressplotInterface
334  o_tsvc=NeEstimationTableFileManager
335 
336  ls_column_names_to_test=o_myc.GROUP_BY_COLUMNS + o_myc.Y_AXIS_VALUE_COLUMNS
337 
338  o_tsv=open( s_tsv_file_name, 'r' )
339  s_header=o_tsv.readline()
340  o_tsv.close()
341 
342  ls_header_fields = s_header.strip().split( o_tsvc.DELIM_TABLE )
343 
344  for s_name in ls_column_names_to_test:
345 
346  if s_name not in ls_header_fields:
347  b_verdict=False
348  break
349  #end if column name missing
350  #end for each column name to test
351 
352  return b_verdict

355  def __looks_like_a_tsv_file_with_data_lines( self, s_file_name ):
356  '''
357  Another simple heuristic added to the header check
358  in __looks_like_tsv_file, to avoid loading
359  empty *tsv files i.e. header is correct
360  but no data lines follow.
361  '''
362  b_return_value=False
363 
364  if self.__looks_like_tsv_file( s_file_name ) == True:
365  o_tsvc=NeEstimationTableFileManager
366 
367  o_file=open( s_file_name, 'r' )
368  #read the header:
369  s_header=o_file.readline()
370  ls_vals=s_header.split( o_tsvc.DELIM_TABLE )
371  i_num_fields=len( ls_vals )
372  ls_second_line=o_file.readline()
373 
374  if ls_second_line is not None:
375  ls_second_line_values=ls_second_line.split( o_tsvc.DELIM_TABLE )
376  i_num_values=len( ls_second_line_values )
377  if i_num_values == i_num_fields:
378  b_return_value=True
379  #end if value count equals field count
380  #end if we have a second line
381  #end if the file looks like a tsv file
