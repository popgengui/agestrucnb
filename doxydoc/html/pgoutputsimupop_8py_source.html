242  def bz2CompressAllFiles(self, ls_files_to_skip=[] ):
243  '''
244  used code and advice in,
245  http://stackoverflow.com/questions/9518705/big-file-compression-with-python
246 
247  Note: checked the shutil documentation at https://docs.python.org/2/library/shutil.html
248  which warns of loss of meta file info (owner/group ACLs) when using shutil.copy() or shutil.copy2(). Unclear
249  whether this applies to the copyfileobj, though my few tests showd all of these were retained.
250  '''
251  for s_myfile in [ self.__outname, self.__errname, self.__megadbname, self.__confname, self.__genepopname ]:
252 
253  if s_myfile in ls_files_to_skip:
254  pass
255  elif not self.__file_exists( s_myfile ):
256  self.__raise_file_not_found_error( s_myfile, "compress file with bz2" )
257  else:
258  with open( s_myfile, 'rb' ) as o_input:
259  with bz2.BZ2File( s_myfile + '.bz2', 'wb', compresslevel=9 ) as o_output:
260  #try/except overkill, but want logic to show
261  #we don't remove the input file
262  #if the copy op fails:
263  try:
264  shutil.copyfileobj( o_input, o_output )
265  except Exception as e:
266  raise e
267  #end try except
268 
269  #required by windows only, else error can't remove:
270  o_input.close()
271  os.remove( s_myfile )
272  #end with bzfile for writing
273  #end with current file for reading
274  #end if file absent, else exists
275  #end for each file

329  '''
330  reads the *.gen file from the simuPop output
331  and produces a genepop file based on the gen file
332 
333  param idx_allele_start: int, one-based index giving the
334  item (loci) number of the first allele entry in the line of gen-file input
335  Example, if the *gen file has 10 microsates and 40 SNPs, then its first
336  10 allele entries (that follow the indiv number and the gneration number)
337  will be the microsats, and the final 40 allele entries will be the SNPs
338  If we wanted only the Microsats to be included in the gen file, then
339  this param would be "1", and the idx_allele_stop value would be 10.
340  If we want both msats and snps, we'd give 1 and 50 for these params.
341  If we want only SNPs, wed enter 11 for start allele and 50 for stop.
342 
343  param idx_allele_stop: int, one-based index giving the
344  item (loci) number of the last allele entry in the line of gen-file input
345 
346  param optional b_do_compress, if true, will bzip2 the output genepop file
347 
348  param optional b_pop_per_gen, if true, will insert a "Pop" line between the
349  generations, as given by the 2nd field in the gen file
350 
351  2017_02_12. In implementing a bias adjustment in the NeEstimator's ldne-based
352  Nb estimates, calculated in pgdriveneestimator.py, We add to this def an arg
353  "f_nbne_ratio", defalut None. If arg is non-None, we append to the header line
354  text, "nbne=str(f_nbne_ratio)". When the GUI interface for the Nb estimation reads
355  in genepop files, it will check for they key=value string and then, if present,
356  pass the value on to the calls that eventually invoke the pgdriveneestimator.
357  '''
